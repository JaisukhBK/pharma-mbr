async function authenticate(req, res, next) {
  const h = req.headers.authorization;
  if (!h || !h.startsWith('Bearer ')) return res.status(401).json({ error: 'Authentication required' });
  try {
    const d = jwt.verify(h.split(' ')[1], SECRET);
    req.session = { userId: d.userId, email: d.email, fullName: d.fullName, role: d.role, permissions: d.permissions || [] };
    next();
  } catch (err) {
    return res.status(401).json({ error: err.name === 'TokenExpiredError' ? 'Token expired' : 'Invalid token' });
  }
}
